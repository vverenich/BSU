.MODEL SMALL
.CODE
START:
	MOV AX,ES:[2CH]
	MOV DS,AX
	MOV SI,-1
	
SEARCH_01:
	ADD SI,1
	MOV AL,[SI]
	CMP AL,0
	JNE SEARCH_01
	MOV AL,[SI+1]
	CMP AL,1
	JNE SEARCH_01
	ADD SI,2
	
SEARCH_START:
	ADD SI,1
	MOV AL,[SI]
	CMP AL,'\'
	JNE SKIP_SAVE
	MOV DI,SI
	
SKIP_SAVE:
	CMP AL,0
	JNE SEARCH_START
	
PRINT_NAME:
	ADD DI,1
	MOV AL,[DI]
	CMP AL,0
	JE FINISH
	INT 29H
	JMP PRINT_NAME

FINISH:
	MOV AX,4C00H
	INT 21H
END START
